## 背景知识

### 为什么需要获取视频直链？

很多在线教育平台的视频回放页面使用动态加载技术（如 Vue.js、React），视频地址不会直接显示在 HTML 源码中，而是通过 JavaScript 动态请求获取。这就导致：

- 无法直接通过查看源码找到视频地址
- 传统的爬虫工具（如 `requests`、`urllib`）无法获取
- 需要模拟真实浏览器行为

### 什么是视频直链？

视频直链是指直接指向视频文件（如 `.mp4`、`.m3u8`）的 URL，例如：
```
https://example.com/path/to/video.mp4
```

有了直链后，可以使用任何下载工具（IDM、wget、curl、yt-dlp 等）直接下载。

## 工具介绍

### Playwright 是什么？

Playwright 是由 Microsoft 开发的浏览器自动化工具，支持：
- **多浏览器**：Chromium（Chrome）、Firefox、WebKit（Safari）
- **多语言**：Python、JavaScript、Java、C#
- **强大的 API**：页面操作、网络拦截、截图、录屏等

### 为什么选择 Playwright？

| 特性 | 传统爬虫 (requests) | Selenium | Playwright |
|------|-------------------|----------|-----------|
| 执行 JavaScript | ❌ | ✅ | ✅ |
| 速度 | 快 | 较慢 | 快 |
| 现代网站支持 | 差 | 一般 | 优秀 |
| 网络拦截 | ❌ | 复杂 | 简单 |
| 无头模式 | - | ✅ | ✅ |

## 完整代码：获取视频直链

创建文件 `get_video_links.py`：

```python
#!/usr/bin/env python3
"""
自动获取视频直链的脚本
使用 Playwright 拦截网络请求
"""

import asyncio
import json
import sys
from playwright.async_api import async_playwright

# 设置编码（Windows 中文输出需要）
sys.stdout.reconfigure(encoding='utf-8')

# 需要获取的视频页面列表
VIDEO_URLS = [
    ("第一节课下午", "https://chgdegabxaa.kekpwl.cn/c/7NXTMn3"),
    ("3月21日上午", "https://chgdegabxaa.kekpwl.cn/c/ZFvrrw3"),
    ("3月21日下午", "https://chgdegabxaa.kekpwl.cn/c/Fr6Rrw3"),
    ("3月28日上午", "https://chgdegabxaa.kekpwl.cn/c/7ngrhp3"),
    ("3月28日下午", "https://chgdegabxaa.kekpwl.cn/c/9qaarzhp3"),
]

# 存储结果
current_video_name = None


async def get_video_url(name, url):
    """
    获取单个视频的直链

    参数:
        name: 视频名称（用于标识）
        url: 视频页面URL
    """
    global current_video_name
    current_video_name = name

    print(f"\n{'='*60}")
    print(f"正在获取: {name}")
    print(f"页面URL: {url}")
    print('='*60)

    async with async_playwright() as p:
        # 1. 启动浏览器（无头模式）
        browser = await p.chromium.launch(headless=True)

        # 2. 创建浏览器上下文（设置 User-Agent）
        context = await browser.new_context(
            user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        )

        # 3. 创建新页面
        page = await context.new_page()

        # 4. 存储找到的视频链接
        found_urls = []

        # 5. 设置响应监听器
        async def handle_response(response):
            """监听所有网络响应，筛选视频链接"""
            resp_url = response.url

            # 检查是否包含视频文件特征
            if '.mp4' in resp_url:
                print(f"[OK] 找到视频直链: {resp_url}")
                found_urls.append(resp_url)

            # 查找包含视频信息的 API 响应
            if any(keyword in resp_url for keyword in ['api', 'vod', 'play']):
                try:
                    content_type = response.headers.get('content-type', '')
                    if 'json' in content_type:
                        body = await response.body()
                        text = body.decode('utf-8', errors='ignore')
                        if '.mp4' in text:
                            print(f"[API] {resp_url}")
                            print(f"内容: {text[:500]}")
                except:
                    pass

        # 注册响应监听器
        page.on("response", lambda response: asyncio.create_task(handle_response(response)))

        try:
            # 6. 访问页面
            await page.goto(url, wait_until='networkidle', timeout=60000)

            # 等待页面加载完成
            await asyncio.sleep(3)

            # 7. 尝试点击播放按钮（触发视频加载）
            play_selectors = [
                'video',
                '.play-btn',
                '[class*="play"]',
                'button',
                '.vjs-big-play-button'
            ]

            for selector in play_selectors:
                try:
                    element = await page.query_selector(selector)
                    if element:
                        await element.click()
                        print(f"[点击] {selector}")
                        await asyncio.sleep(2)
                        break
                except:
                    continue

            # 8. 等待视频加载和网络请求完成
            await asyncio.sleep(8)

            # 9. 从页面 DOM 中提取视频地址
            video_elements = await page.query_selector_all('video')
            print(f"[信息] 找到 {len(video_elements)} 个 video 元素")

            for i, video in enumerate(video_elements):
                # 获取 video 标签的 src 属性
                src = await video.get_attribute('src')
                if src:
                    print(f"[OK] video[{i}] src: {src}")
                    found_urls.append(src)

                # 获取 source 子标签的 src
                sources = await video.query_selector_all('source')
                for j, source in enumerate(sources):
                    src = await source.get_attribute('src')
                    if src:
                        print(f"[OK] source[{j}] src: {src}")
                        found_urls.append(src)

            # 10. 通过 JavaScript 获取更多视频信息
            js_video_urls = await page.evaluate('''() => {
                const urls = [];
                document.querySelectorAll('video').forEach(v => {
                    if (v.src) urls.push(v.src);
                    v.querySelectorAll('source').forEach(s => {
                        if (s.src) urls.push(s.src);
                    });
                });
                return urls;
            }''')

            for vurl in js_video_urls:
                print(f"[JS] 找到: {vurl}")

        except Exception as e:
            print(f"错误: {e}")

        # 11. 关闭浏览器
        await browser.close()

        return found_urls[0] if found_urls else None


async def main():
    """主函数：批量处理所有视频"""
    print("开始自动获取视频直链...")
    print(f"共 {len(VIDEO_URLS)} 个视频")

    results = {}

    for name, url in VIDEO_URLS:
        link = await get_video_url(name, url)
        if link:
            results[name] = link

    # 输出结果汇总
    print("\n" + "="*60)
    print("获取结果汇总:")
    print("="*60)

    for name, url in results.items():
        print(f"\n{name}:")
        print(f"  {url}")

    # 保存到文件
    with open('video_links.txt', 'w', encoding='utf-8') as f:
        for name, url in results.items():
            f.write(f"{name}\n{url}\n\n")

    print("\n结果已保存到 video_links.txt")


if __name__ == "__main__":
    asyncio.run(main())
```

## 代码详解

### 1. 响应监听器的核心逻辑

```python
async def handle_response(response):
    resp_url = response.url

    # 检测视频文件链接
    if '.mp4' in resp_url:
        print(f"找到视频: {resp_url}")
        found_urls.append(resp_url)
```

当浏览器发送任何请求时，Playwright 会拦截响应。我们通过检查 URL 是否包含 `.mp4` 来识别视频链接。

### 2. 页面操作

```python
# 访问页面
await page.goto(url, wait_until='networkidle', timeout=60000)
```

- `wait_until='networkidle'`：等待网络空闲（没有新的网络请求）
- `timeout=60000`：超时时间 60 秒

### 3. 点击播放按钮

```python
element = await page.query_selector('.play-btn')
if element:
    await element.click()
```

很多视频需要点击播放按钮后才会加载真实视频地址。

### 4. 执行 JavaScript

```python
js_video_urls = await page.evaluate('''() => {
    const urls = [];
    document.querySelectorAll('video').forEach(v => {
        if (v.src) urls.push(v.src);
    });
    return urls;
}''')
```

在页面上下文中执行 JS，可以获取动态加载的内容。

## 原理解析

### 视频加载流程

```
用户访问页面 → 加载HTML/JS → JS请求视频信息API → 返回视频直链 → 播放器加载视频
     ↑                                                       ↓
     └──────────────── Playwright监听响应 ←─────────────────┘
```

Playwright 在浏览器层面拦截所有网络请求，因此可以捕获动态加载的视频地址。

### 关键技巧

| 技巧 | 作用 |
|-----|------|
| 响应监听 | 捕获所有网络请求，筛选视频链接 |
| 点击播放 | 触发视频加载，生成真实请求 |
| 等待时间 | 给 JavaScript 足够时间执行 |
| DOM查询 | 从页面元素中提取视频地址 |
| JS执行 | 访问页面内部变量和数据 |

## 常见问题

### Q1: 找不到视频链接？

**可能原因：**
- 视频需要登录才能观看
- 需要等待更长时间
- 视频使用 m3u8 格式而非 mp4

**解决方案：**
```python
# 添加更多等待时间
await asyncio.sleep(15)

# 检测 m3u8
if '.m3u8' in resp_url or '.mp4' in resp_url:
    found_urls.append(resp_url)
```

### 2. 批量下载脚本 (download_videos.py)

```python
#!/usr/bin/env python3
"""
批量下载视频的简化脚本
支持 yt-dlp（推荐）和 curl 两种下载方式
"""

import subprocess
import os
import shutil

# 视频链接（从 video_links.txt 复制或读取）
VIDEOS = {
    "第一节课下午": "https://111453136245362688.tenwiseacademy.cn/.../b3b19f6c...mp4",
    "3月21日上午": "https://111453136245362688.tenwiseacademy.cn/.../e356227d...mp4",
    # ... 更多视频
}


def download_with_ytdlp():
    """使用 yt-dlp 下载（推荐，支持断点续传）"""
    print("使用 yt-dlp 下载视频...")

    for name, url in VIDEOS.items():
        filename = f"{name}.mp4"
        print(f"\n正在下载: {name}")

        cmd = [
            "yt-dlp",
            "-o", filename,
            "--no-check-certificate",
            url
        ]

        try:
            subprocess.run(cmd, check=True)
            print(f"[OK] {name} 下载完成")
        except subprocess.CalledProcessError as e:
            print(f"[错误] {name} 下载失败: {e}")


def download_with_curl():
    """使用 curl 下载（备用方案）"""
    print("使用 curl 下载视频...")

    for name, url in VIDEOS.items():
        filename = f"{name}.mp4"
        print(f"\n正在下载: {name}")

        cmd = [
            "curl",
            "-L",  # 跟随重定向
            "-o", filename,
            "--progress-bar",
            url
        ]

        try:
            subprocess.run(cmd, check=True)
            print(f"[OK] {name} 下载完成")
        except subprocess.CalledProcessError as e:
            print(f"[错误] {name} 下载失败: {e}")


def check_tools():
    """检查可用的下载工具"""
    if shutil.which("yt-dlp"):
        return "yt-dlp"
    elif shutil.which("curl"):
        return "curl"
    else:
        return None


if __name__ == "__main__":
    # 创建下载目录
    download_dir = "课程回放"
    os.makedirs(download_dir, exist_ok=True)
    os.chdir(download_dir)

    print(f"视频将下载到: {os.path.abspath('.')}")

    # 检查工具并下载
    tool = check_tools()
    if tool == "yt-dlp":
        download_with_ytdlp()
    elif tool == "curl":
        download_with_curl()
    else:
        print("错误: 未找到 yt-dlp 或 curl")
        print("请先安装 yt-dlp: pip install yt-dlp")

    print("\n全部下载完成!")
```

### 3. 下载工具对比

| 工具 | 优点 | 缺点 | 适用场景 |
|-----|------|------|---------|
| **yt-dlp** | 自动重试、断点续传、自动合并 | 需要安装 | 大文件、网络不稳定 |
| **curl** | 系统自带、轻量 | 无断点续传 | 小文件、快速下载 |
| **wget** | 支持递归下载 | Windows需安装 | Linux服务器 |
| **IDM** | 图形界面、多线程 | 需付费 | 日常使用 |
| **aria2** | 超高速、多连接 | 配置复杂 | 批量高速下载 |

### 核心步骤

使用 Playwright 获取并下载视频的完整流程：

1. **环境准备** - 安装 Python、Playwright、浏览器
2. **启动浏览器** - 创建无头或有界面浏览器实例
3. **监听网络** - 拦截所有响应，筛选视频链接
4. **访问页面** - 加载目标视频页面
5. **触发加载** - 点击播放按钮等操作
6. **提取链接** - 从响应和 DOM 中获取视频地址
7. **保存结果** - 输出到 video_links.txt
8. **批量下载** - 使用 yt-dlp/curl 下载所有视频

### 适用场景

这种方法适用于大多数现代视频网站：
- ✓ Vue/React 单页应用
- ✓ 视频加密/防盗链
- ✓ m3u8 流媒体
- ✓ 需要登录/ Cookie 验证
- ✓ 动态加载/懒加载

### 关键技术点

| 技术 | 应用场景 |
|-----|---------|
| Playwright | 浏览器自动化、JS执行 |
